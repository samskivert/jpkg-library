031    package com.threerings.antidote.field.text;
032    
033    import static com.threerings.antidote.MutabilityHelper.requiresValidation;
034    
035    /**
036     * A {@link TextField} that enforces the text data is always set, not zero length, and has
037     * no newlines or tabs.
038     */
039    public abstract class SingleLineTextField extends TextField
040    {
041        @Override // from BaseComponent
042        protected final void validateField ()
043        {
044            // require the text was set, and was not empty.
045            if (validateTextNotEmpty() == TextStatus.INVALID_TEXT) return;
046            // scrub any newlines or tabs from the field.
047            scrubTextWhitespace();
048    
049            _validatedText = super.getText();
050            validateTextField();
051        }
052    
053        /**
054         * Override getText() so that the initial steps of validateField() must have happened before
055         * access.
056         */
057        @Override // from TextField
058        protected String getText()
059        {
060            requiresValidation(_validatedText);
061            return _validatedText;
062        }
063    
064        /**
065         * Provide concrete classes a method for additional validation.
066         */
067        protected abstract void validateTextField ();
068    
069        /** The text after being validated. */
070        private String _validatedText;
071    }
