031    package com.threerings.jpkg.ant.dpkg.info;
032    
033    import com.threerings.antidote.field.text.EnumTextField;
034    import com.threerings.jpkg.debian.PackagePriority;
035    
036    /**
037     * Stores the &lt;info&gt; &lt;priority&gt; field, which is the package priority.
038     * @see PackagePriority
039     */
040    public class Priority extends EnumTextField<PackagePriority>
041    {
042        public Priority ()
043        {
044            super(PackagePriority.class);
045        }
046    
047        // from Field
048        public String getFieldName ()
049        {
050            return "priority";
051        }
052    
053        /**
054         * Returns the user data converted into a {@link PackagePriority}. Cannot be called before validate().
055         */
056        public PackagePriority getPackagePriority ()
057        {
058            return getEnum();
059        }
060    }
