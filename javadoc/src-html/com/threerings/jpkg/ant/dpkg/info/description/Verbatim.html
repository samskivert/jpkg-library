031    package com.threerings.jpkg.ant.dpkg.info.description;
032    
033    import java.util.ArrayList;
034    import java.util.Collections;
035    import java.util.List;
036    
037    import com.threerings.antidote.Violation;
038    import com.threerings.antidote.field.text.SimpleTextField;
039    import com.threerings.jpkg.ant.dpkg.info.ControlDataViolation;
040    import com.threerings.jpkg.debian.ControlDataInvalidException;
041    import com.threerings.jpkg.debian.PackageDescription;
042    
043    public class Verbatim extends SimpleTextField
044        implements DescriptionAction
045    {
046        // from Field
047        public String getFieldName ()
048        {
049            return "verbatim";
050        }
051    
052        // from DescriptionAction
053        public List<Violation> apply (PackageDescription description)
054        {
055            try {
056                description.addParagraph(getText());
057    
058            } catch (final ControlDataInvalidException cdie) {
059                final List<Violation> violations = new ArrayList<Violation>();
060                violations.add(new ControlDataViolation(this, cdie));
061                return violations;
062            }
063    
064            return Collections.emptyList();
065        }
066    }
