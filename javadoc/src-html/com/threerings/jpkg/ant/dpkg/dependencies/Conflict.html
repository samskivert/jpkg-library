031    package com.threerings.jpkg.ant.dpkg.dependencies;
032    
033    import com.threerings.antidote.property.StringProperty;
034    import com.threerings.jpkg.ant.dpkg.dependencies.conditions.Condition;
035    import com.threerings.jpkg.debian.PackageInfo;
036    import com.threerings.jpkg.debian.dependency.PackageConflict;
037    
038    /**
039     * The &lt;dependencies&gt; &lt;conflict&gt; field. Indicates a package conflict.
040     */
041    public class Conflict extends BaseDependency<PackageConflict>
042    {
043        // from Field
044        public String getFieldName ()
045        {
046            return "conflict";
047        }
048    
049        // from PackageInfoDependency
050        public void addToPackageInfo (PackageInfo info)
051        {
052            info.addConflict(getDependency());
053        }
054    
055        @Override // from BaseDependency
056        protected PackageConflict createDependency (StringProperty packageName)
057        {
058            return new PackageConflict(packageName.getValue());
059        }
060    
061        @Override // from BaseDependency
062        protected PackageConflict createDependency (StringProperty packageName, Condition condition)
063        {
064            return new PackageConflict(packageName.getValue(), condition.getVersion(), condition.getRelationship());
065        }
066    }
