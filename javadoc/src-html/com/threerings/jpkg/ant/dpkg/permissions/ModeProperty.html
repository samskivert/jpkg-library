031    package com.threerings.jpkg.ant.dpkg.permissions;
032    
033    import com.threerings.antidote.field.Field;
034    import com.threerings.antidote.property.BaseProperty;
035    
036    /**
037     * A {@link BaseProperty} that holds an {@link Integer} object meant to represent a chmod(1) file
038     * mode in octal.
039     */
040    public class ModeProperty extends BaseProperty<Integer>
041    {
042        // from BaseProperty
043        public ModeProperty (String name, Field field)
044        {
045            super(name, field);
046        }
047    
048        @Override // from BaseProperty
049        protected Integer validateProperty ()
050        {
051            try {
052                return Integer.parseInt(getRawValue(), OCTAL);
053    
054            } catch (final NumberFormatException nfe) {
055                appendViolation(new InvalidModeViolation(this, getRawValue()));
056                return null;
057            }
058        }
059    
060        /** The radix for octal, which is what the file permissions are expected to be in. */
061        private static final int OCTAL = 8;
062    }
